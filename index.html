Importing and cleaning the dataset:

data(ny_noaa)

set.seed(123) # I randomly choose 1000 lines because the dataset is too large
ny_noaa <- ny_noaa %>%
  sample_n(1000)

ny_noaa <- ny_noaa %>%
  filter(!is.na(prcp), !is.na(snow))

Scatter Plot

ny_noaa %>%
  mutate(text_label = str_c("Date: ", date, "\nPrecipitation: ", prcp)) %>% 
  plot_ly(
    x = ~date, y = ~prcp, type = "scatter", mode = "markers",
    color = ~prcp, text = ~text_label, alpha = 0.5)

Boxplot

ny_noaa %>%
  mutate(date = as.factor(date)) %>% 
  plot_ly(y = ~snow, color = ~date, type = "box", colors = "viridis")

Bar Chart

ny_noaa_count <- ny_noaa %>%
  group_by(date) %>%
  summarize(n = n())

plot_ly(data = ny_noaa_count, x = ~date, y = ~n, type = 'bar', colors = 'viridis') %>%
  layout(title = "Bar Chart of Dates", xaxis = list(title = "Date"), yaxis = list(title = "Count"))
